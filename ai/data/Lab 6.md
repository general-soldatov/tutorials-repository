    r = np.linspace(0.0, 1, N)
    t = np.linspace(j * 4, (j + 1) * 4, N) + np.random.randn(N) * 0.2 # theta
    X[ix] = np.c_[r * np.sin(t), r * np.cos(t)]
    y[ix] = j
```

# D_in - —Ä–∞–∑–º–µ—Ä–Ω–æ—Å—Ç—å –≤—Ö–æ–¥–∞ (–∫–æ–ª–∏—á–µ—Å—Ç–≤–æ –ø—Ä–∏–∑–Ω–∞–∫–æ–≤ —É –æ–±—ä–µ–∫—Ç–∞)
# H - —Ä–∞–∑–º–µ—Ä–Ω–æ—Å—Ç—å —Å–∫—Ä—ã—Ç—ã—Ö —Å–ª–æ—ë–≤
# D_out - —Ä–∞–∑–º–µ—Ä–Ω–æ—Å—Ç—å –≤—ã—Ö–æ–¥–Ω–æ–≥–æ —Å–ª–æ—è (—Å—É—Ç—å - –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ –∫–ª–∞—Å—Å–æ–≤)

    torch.nn.Linear(D_in, H),
    torch.nn.ReLU(),
    torch.nn.Linear(H, D_out),
)

optimizer = torch.optim.SGD(two_layer_net.parameters(), lr=learning_rate)

    y_pred = two_layer_net(X)

    # Loss
    loss = loss_fn(y_pred, y)

    optimizer.zero_grad()

    # backward
    loss.backward()

    # UPDATE!
    optimizer.step()
```
